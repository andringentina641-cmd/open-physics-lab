services:
  wg-easy:
    image: ghcr.io/wg-easy/wg-easy:15
    container_name: wg-easy
    environment:
      
      # Öffentliche Domain oder DDNS
      - WG_HOST=gbssg.ddns.net
      
      # WireGuard UDP Port
      - WG_PORT=51824  # This is correct for your DDNS/Router
      
      # Web UI Port
      - PORT=51821
      
      # Zeitzone
      - TZ=Europe/Zurich
      
      # Optional: DNS für Clients
      - WG_DEFAULT_DNS=1.1.1.1,8.8.8.8
      
      # Optional: Standard-IP-Bereich
      - WG_DEFAULT_ADDRESS=10.8.0.x
      
      # IPv6 deaktivieren
      - DISABLE_IPV6=true
      
      # Setup über WebUI
      - INSECURE=true

    volumes:
      - wg_easy_data:/etc/wireguard

    ports:
      - "51820:51824/udp"  # Map Pi's 51820 to Container's 51832 
      - "51821:51821/tcp"   #WebUI

    restart: unless-stopped

    cap_add:
      - NET_ADMIN
      - SYS_MODULE

    sysctls:
      - net.ipv4.ip_forward=1
      - net.ipv4.conf.all.src_valid_mark=1

volumes:
  wg_easy_data:
